## 1. 환경 매트릭스

| 항목 | Local | Dev | Staging | Prod |
|------|-------|-----|---------|------|
| **목적** | 개발/디버깅 | 통합 테스트 | QA/릴리즈 후보 | 실서비스 |
| **DB** | Docker PG | RDS (db.t3.micro) | RDS (db.t3.small) | RDS (db.r6g.large) |
| **Redis** | Docker Redis | ElastiCache (t3.micro) | ElastiCache (t3.small) | ElastiCache (r6g.large) Cluster |
| **ES** | Docker ES | OpenSearch (t3.small) | OpenSearch (m6g.large) | OpenSearch (r6g.large) |
| **Kafka** | Docker Kafka | MSK (kafka.t3.small) | MSK (kafka.m5.large) | MSK (kafka.m5.xlarge) |
| **K8s** | Docker Compose | EKS (t3.medium x2) | EKS (m5.large x2) | EKS (m5.xlarge x3) |
| **도메인** | localhost | dev.synapse.app | staging.synapse.app | synapse.app |
| **TLS** | mkcert (self-signed) | ACM | ACM | ACM + Cloudflare |
| **로그** | Console | CloudWatch | CloudWatch | CloudWatch + Datadog |
| **모니터링** | - | Prometheus/Grafana | Prometheus/Grafana | Datadog APM |

## 3. Docker Compose (로컬 개발)

```yaml
# docker-compose.yml
version: "3.9"

services:
  # ===== PostgreSQL 16 + pgvector =====
  postgres:
    image: pgvector/pgvector:pg16
    container_name: synapse-postgres
    environment:
      POSTGRES_DB: synapse
      POSTGRES_USER: synapse_user
      POSTGRES_PASSWORD: changeme_db_password
    ports:
      - "5432:5432"
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./infra/docker/postgres/init.sql:/docker-entrypoint-initdb.d/01-init.sql
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U synapse_user -d synapse"]
      interval: 10s
      timeout: 5s
      retries: 5

  # ===== Redis 7 =====
  redis:
    image: redis:7-alpine
    container_name: synapse-redis
    command: redis-server --requirepass changeme_redis_password --maxmemory 256mb --maxmemory-policy allkeys-lru
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    healthcheck:
      test: ["CMD", "redis-cli", "-a", "changeme_redis_password", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5

  # ===== Elasticsearch 8 + Nori =====
  elasticsearch:
    image: docker.elastic.co/elasticsearch/elasticsearch:8.12.0
    container_name: synapse-elasticsearch
    environment:
      - discovery.type=single-node
      - xpack.security.enabled=false
      - ES_JAVA_OPTS=-Xms512m -Xmx512m
      - cluster.name=synapse-es
    ports:
      - "9200:9200"
    volumes:
      - es_data:/usr/share/elasticsearch/data
    healthcheck:
      test: ["CMD-SHELL", "curl -f http://localhost:9200/_cluster/health || exit 1"]
      interval: 15s
      timeout: 10s
      retries: 5
    # Nori 플러그인 설치 (빌드 시)
    # 커스텀 Dockerfile 사용 권장:
    # FROM docker.elastic.co/elasticsearch/elasticsearch:8.12.0
    # RUN bin/elasticsearch-plugin install analysis-nori

  # ===== Kafka + Zookeeper =====
  zookeeper:
    image: confluentinc/cp-zookeeper:7.5.0
    container_name: synapse-zookeeper
    environment:
      ZOOKEEPER_CLIENT_PORT: 2181
      ZOOKEEPER_TICK_TIME: 2000
    ports:
      - "2181:2181"

  kafka:
    image: confluentinc/cp-kafka:7.5.0
    container_name: synapse-kafka
    depends_on:
      - zookeeper
    environment:
      KAFKA_BROKER_ID: 1
      KAFKA_ZOOKEEPER_CONNECT: zookeeper:2181
      KAFKA_ADVERTISED_LISTENERS: PLAINTEXT://localhost:9092
      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
      KAFKA_AUTO_CREATE_TOPICS_ENABLE: "true"
    ports:
      - "9092:9092"
    healthcheck:
      test: ["CMD-SHELL", "kafka-broker-api-versions --bootstrap-server localhost:9092 || exit 1"]
      interval: 15s
      timeout: 10s
      retries: 5

  # ===== Mailhog (로컬 메일 테스트) =====
  mailhog:
    image: mailhog/mailhog:latest
    container_name: synapse-mailhog
    ports:
      - "1025:1025"   # SMTP
      - "8025:8025"   # Web UI
    environment:
      MH_STORAGE: memory

  # ===== Kafka UI (개발 편의) =====
  kafka-ui:
    image: provectuslabs/kafka-ui:latest
    container_name: synapse-kafka-ui
    depends_on:
      - kafka
    environment:
      KAFKA_CLUSTERS_0_NAME: synapse-local
      KAFKA_CLUSTERS_0_BOOTSTRAPSERVERS: kafka:9092
    ports:
      - "8090:8080"

volumes:
  postgres_data:
  redis_data:
  es_data:
```

## 4. 초기화 SQL (PostgreSQL)

```sql
-- infra/docker/postgres/init.sql

-- pgvector 확장 활성화
CREATE EXTENSION IF NOT EXISTS vector;
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- RLS 활성화를 위한 기본 설정
ALTER DATABASE synapse SET app.current_tenant_id = '';

-- 테넌트 격리를 위한 공통 함수
CREATE OR REPLACE FUNCTION current_tenant_id()
RETURNS UUID AS $$
  SELECT NULLIF(current_setting('app.current_tenant_id', true), '')::UUID;
$$ LANGUAGE SQL STABLE;
```

## 6. 환경별 시작 명령어

| 환경 | 명령어 |
|------|--------|
| 인프라 기동 | `docker compose up -d` |
| Backend (Spring) | `./gradlew bootRun --args='--spring.profiles.active=local'` |
| AI Service (FastAPI) | `cd ai-service && uvicorn main:app --reload --port 8000` |
| Frontend (Flutter Web) | `cd frontend && flutter run -d chrome --dart-define=ENV=local` |
| 전체 종료 | `docker compose down` |
| 전체 초기화 | `docker compose down -v && docker compose up -d` |
